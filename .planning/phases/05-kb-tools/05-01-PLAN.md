2. **Create lib/kb-tools.ts** with:

- `kbExplore()`: Query all published items, aggregate by category and type client-side. Return `{ categories, types, total }`.

- `kbList(params)`: Filter by optional `category`, `type`, `limit` (default 10). Return array of `{ slug, title, summary, category, type }`.

- `kbRead(slug)`: Get single item by slug. Return `{ slug, title, type, category, content }` or null.

- `kbSearch(query, limit)`: 
  - First try `textSearch('title', query, { type: 'websearch' })`
  - If no results, fallback to ILIKE on title, summary, content
  - Return array of `{ slug, title, summary, category, type }`

3. **Export KB_TOOLS array** (Tool definitions for Anthropic API):
```typescript
import { Tool } from '@anthropic-ai/sdk/resources/messages'

export const KB_TOOLS: Tool[] = [
  { name: "kb_explore", description: "...", input_schema: {...} },
  { name: "kb_list", description: "...", input_schema: {...} },
  { name: "kb_read", description: "...", input_schema: {...} },
  { name: "kb_search", description: "...", input_schema: {...} }
]
```

4. **Export executeTool dispatcher**:
```typescript
export async function executeTool(name: string, input: Record<string, unknown>): Promise<string> {
  switch (name) {
    case 'kb_explore': return JSON.stringify(await kbExplore())
    case 'kb_list': return JSON.stringify(await kbList(input))
    case 'kb_read': return JSON.stringify(await kbRead(input.slug as string))
    case 'kb_search': return JSON.stringify(await kbSearch(input.query as string, input.limit as number))
    default: return JSON.stringify({ error: 'Unknown tool' })
  }
}
```

**Implementation notes from 05-RESEARCH.md:**
- Use `createServerClient()` from `lib/supabase.ts` (Service Role, bypasses RLS)
- Always filter by `status: 'published'`
- Tool results must be JSON.stringify'd (Anthropic API expects strings)
- Tool descriptions should be 3-4 sentences explaining when to use each tool

```typescript
// scripts/test-kb-tools.ts
import { kbExplore, kbList, kbRead, kbSearch } from '../lib/kb-tools'

async function main() {
  console.log('--- Testing KB Tools ---\n')
  
  // Test kb_explore
  console.log('1. kb_explore()')
  const explore = await kbExplore()
  console.log('   Categories:', Object.keys(explore.categories).length)
  console.log('   Types:', Object.keys(explore.types).length)
  console.log('   Total:', explore.total)
  if (explore.total === 0) throw new Error('kb_explore returned no items')
  
  // Test kb_list
  console.log('\n2. kb_list({ type: "tool", limit: 3 })')
  const list = await kbList({ type: 'tool', limit: 3 })
  console.log('   Found:', list.length, 'items')
  list.forEach(item => console.log('   -', item.title))
  
  // Test kb_read (use first item from list)
  if (list.length > 0) {
    console.log('\n3. kb_read("' + list[0].slug + '")')
    const item = await kbRead(list[0].slug)
    if (!item) throw new Error('kb_read returned null')
    console.log('   Title:', item.title)
    console.log('   Content length:', item.content.length, 'chars')
  }
  
  // Test kb_search
  console.log('\n4. kb_search("ChatGPT")')
  const search = await kbSearch('ChatGPT', 3)
  console.log('   Found:', search.length, 'items')
  search.forEach(item => console.log('   -', item.title))
  
  console.log('\n--- All tests passed ---')
}

main().catch(err => {
  console.error('\n--- TEST FAILED ---')
  console.error(err)
  process.exit(1)
})
```
